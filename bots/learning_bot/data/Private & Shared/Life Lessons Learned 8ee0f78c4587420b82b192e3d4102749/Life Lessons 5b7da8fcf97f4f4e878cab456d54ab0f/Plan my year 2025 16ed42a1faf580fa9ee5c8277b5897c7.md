    app = Flask(__name__)
    
    # Sample endpoint: Match job descriptions and resumes
    @app.route('/match', methods=['POST'])
    def match():
        data = request.json
        job_description = data.get('job_description', '')
        resume = data.get('resume', '')
    
        # Simple match logic (placeholder)
        match_score = len(set(job_description.split()) & set(resume.split())) / len(set(job_description.split()))
        return jsonify({'match_score': round(match_score * 100, 2)})
    
    if __name__ == '__main__':
        app.run(debug=True)
